\question[Lecture 1]{%
Let \(Q\) be the quadratic form associated with an \(n\times n\) matrix \(A\). Prove that \begin{enumerate}
    \item Q is positive definite \(\iff\) \(\lambda_i > 0\) \(\forall i \in \N\).
    \item Q is negative definite \(\iff\) \(\lambda_i < 0\) \(\forall i \in \N\). 
    \item Q is indefinite \(\iff\) there are positive and negative eigenvalues.
\end{enumerate}
}

    \item A is negative definite \(\iff\) \((-1)^k \cdot det(A_k) > 0 \), \(\forall k \in \N\).
    \item A is indefinite \(\iff\) A is neither positive nor negative definite. 
\end{enumerate}
